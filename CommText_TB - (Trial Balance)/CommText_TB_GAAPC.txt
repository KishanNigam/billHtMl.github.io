 Declare @BASIS varchar(15)= 'GAAPC'
 
 SELECT 
 BD.CAL_ACCTG_PERIOD, 
 BD.COMPANY_CODE,
 BD.ACCOUNT_NUMBER,
 AC.DESCRIPTION, 
 SD2.BU_CODE AS 'SEG',
 sum(BD.CONVERTED_AMOUNT) AS 'BALANCE', 
 BH.STATUS_CODE,
 A.ACCTG_BASIS_CODE

 FROM   
 BSTRN_HEADER BH 
 INNER JOIN BSTRN_DETAIL BD ON BH.TRAN_ID=BD.TRAN_ID 
 INNER JOIN ACCT_BAS_ACCT_BAS A ON BD.ACCTG_BASIS_CODE = A.REL_ACCT_BAS_CODE
 LEFT OUTER JOIN BU_SET_DETAILS SD2 ON BD.BU_SET_ID=SD2.BU_SET_ID AND SD2.BU_TYPE = 'SEG'
 FULL OUTER JOIN ACCOUNT AC ON BD.ACCOUNT_NUMBER=AC.ACCT_NUMBER AND BD.COMPANY_CODE=AC.COMPANY_CODE

 WHERE   
 (BD.CAL_ACCTG_PERIOD>=@PRIOR_PERIOD AND BD.CAL_ACCTG_PERIOD<=@CURRENT_PERIOD) 
 AND A.ACCTG_BASIS_CODE = @BASIS
 AND BD.COMPANY_CODE=@COMPANY_CODE 
 AND BD.ACCOUNT_NUMBER BETWEEN '10000000' AND '39999999' 
 OR (BD.CAL_ACCTG_PERIOD>=@PRIOR_PERIOD2 AND BD.CAL_ACCTG_PERIOD<=@CURRENT_PERIOD) 
 AND A.ACCTG_BASIS_CODE = @BASIS
 AND BD.COMPANY_CODE=@COMPANY_CODE 
 AND BD.ACCOUNT_NUMBER BETWEEN '40000000' AND '59999999'
 OR (BD.CAL_ACCTG_PERIOD>=@PRIOR_PERIOD AND BD.CAL_ACCTG_PERIOD<=@CURRENT_PERIOD) 
 AND A.ACCTG_BASIS_CODE = @BASIS
 AND BD.COMPANY_CODE=@COMPANY_CODE 
 AND BD.ACCOUNT_NUMBER BETWEEN '60000000' AND '63999999' 
 OR (BD.CAL_ACCTG_PERIOD>=@PRIOR_PERIOD2 AND BD.CAL_ACCTG_PERIOD<=@CURRENT_PERIOD) 
 AND A.ACCTG_BASIS_CODE = @BASIS
 AND BD.COMPANY_CODE=@COMPANY_CODE 
 AND BD.ACCOUNT_NUMBER BETWEEN '64000000' AND '69999999'
 OR (BD.CAL_ACCTG_PERIOD>=@PRIOR_PERIOD AND BD.CAL_ACCTG_PERIOD<=@CURRENT_PERIOD) 
 AND A.ACCTG_BASIS_CODE = @BASIS
 AND BD.COMPANY_CODE=@COMPANY_CODE 
 AND BD.ACCOUNT_NUMBER BETWEEN '70000000' AND '79999999'
 OR (BD.CAL_ACCTG_PERIOD>=@PRIOR_PERIOD2 AND BD.CAL_ACCTG_PERIOD<=@CURRENT_PERIOD) 
 AND A.ACCTG_BASIS_CODE = @BASIS
 AND BD.COMPANY_CODE=@COMPANY_CODE 
 AND BD.ACCOUNT_NUMBER BETWEEN '80000000' AND '99999999'


 GROUP BY
 BD.CAL_ACCTG_PERIOD, 
 BD.COMPANY_CODE,
 BD.ACCOUNT_NUMBER,
 AC.DESCRIPTION,
 SD2.BU_CODE,
 BH.STATUS_CODE,
 A.ACCTG_BASIS_CODE

 ORDER BY 
 BD.ACCOUNT_NUMBER,
 BD.CAL_ACCTG_PERIOD
